		var auto = new ajk.Autocomplete($('#search-esf'),{
		        insertDom:$('#search-esf').parent(),
		        template : _.template($('#search-tpl').html()),
		        emptyFetch : true,
		        dataSource: function(keyword,setData){
		            $.ajax({
		                url: op.url,
		                type: 'GET',
		                dataType: 'jsonp',
		                data: {
		                    kw: keyword,
		                    city_id : op.cityId,
		                    app_id: 1,//1:二手房 2:租房 3:新房
							from: 2,//1:首页 2:二手房列表页
							num: 10
		                }
		            }).done(function(data) {
		            	if(keyword){
		            		var kwstr = $.trim(keyword),
		            		kwstr = kwstr.replace(/[~'!<>@#$%^&*()-+_=:]/g, ""),
		            	    regS = new RegExp(kwstr,"g"),
		            	    dataNew = _.toArray(data.data);
			            	for( var i = 0, len = dataNew.length; i < len; ++i  ) {
			            		dataNew[i].keyword = dataNew[i].keyword.replace(regS,'<em class="red">'+kwstr+'</em>');
			            	}
			            	setData(dataNew);
		            	} else {
		            		setData(_.toArray(data.data));
		            	}
		            });
		        }
		    }),
	        close  = $('#search-close'),
	        btn    = $('#search-button');
	    auto.on('inputed',function(kw){
	    	if(!_.isEmpty(kw)){
	    		close.show();
	    	}
	    });
	    auto.on('empty',function(){
	    	close.hide();
	    });
	    close.on('click',function(){
	    	auto.el.val('').focus();
	        ajk.Logger.sendSoj({
	        	site: 'anjuke-npv',
	        	page: close.attr('data-tracker')
	        });
	    });
	    btn.on('click',function(){
	    	window.location.href = op.submitUrl + '?kw=' + auto.el.val() + '&from=zjsr';
	    });
	    auto.on('selected',function(i){
	    	if( i.data.link ) {
                if(i.data.link.indexOf('?') >= 0){ 
                    window.location.href = i.data.link + '&from=xlts_pz';
                } else { 
                    window.location.href = i.data.link + '?from=xlts_pz';
                }
            }else{
                var fromCode='';
                if(i.data.type==2){
                    fromCode = 'xlts_rm';
                }else if(i.data.type==3){
                    fromCode = 'xlts_sg';
                }else if(i.data.type==4){
                    fromCode = 'xlts_kg';
                }else if(i.data.type==5){
                    fromCode = 'xlts_tj';
                }else{
                    if(i.data.isa==1){
                        fromCode = 'xlts_dz';
                    }else{
                        fromCode = 'xlts_mc';
                    }
                    
                }
                window.location.href = op.submitUrl + '?kw=' + auto.el.val() + '&from=' + fromCode;
            }
	    	
	    });


	}

		cityId: "" || 11,
		searchType: '1',
		submitUrl: '/sale/'
	});
