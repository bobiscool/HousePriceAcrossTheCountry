    	var search = location.search;
        var query  = '';
        var params = {};
        search.replace(/\?(.*)/g, function(all, q) {
            query = q;
            return '';
        });
        var arr = query.length ? query.split('&') : [];
        for(var i = 0; i< arr.length; i++){
            var keyVal = arr[i].split('=');
            params[keyVal[0]] = decodeURIComponent(keyVal[1]);
        }

		if( params.kw !== undefined && params.kw != ''){
			$('#search-rent').val(params.kw);
			$('#search-close').show();
		}

		        insertDom:$('#search-rent').parent(),
		        template : _.template($('#search-tpl').html()),
		        dataSource: function(keyword,setData){
		            $.ajax({
		                url: op.url,
		                type: 'GET',
		                dataType: 'jsonp',
		                data: {
		                    kw: keyword,
		                    c : op.cityId,
		                    n : 10,
		                    g : 1,
		                    t : op.searchType
		                }
		            }).done(function(data) {
		                setData(data.list);
		            });
		        }
		    }),
	        close  = $('#search-close'),
	        btn    = $('#search-button');
	    auto.on('inputed',function(kw){
	    	if(!_.isEmpty(kw)){
	    		close.show();
	    	}
	    });
	    auto.on('empty',function(){
	        this.hide();
	    	close.hide();
	    });
	    close.on('click',function(){
	    	auto.el.val('').focus();
	        auto.trigger('empty');
	        ajk.Logger.sendSoj({
	        	site: 'anjuke-npv',
	        	page: close.attr('data-tracker')
	        });
	    });
	    btn.on('click',function(){
