            query = q;

            var keyVal = arr[i].split('=');
            params[keyVal[0]] = decodeURIComponent(keyVal[1]);

		                    c : op.cityId,
		                    n : 10,
		                    g : 1,
		                    t : op.searchType

		            }).done(function(data) {
		                setData(data.list);
		            });
